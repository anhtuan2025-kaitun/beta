local ChillObjects = {}
local ChillActive = false

local function EnableChill()
    local hrp = LocalPlayer.Character:FindFirstChild("HumanoidRootPart")
    Humanoid.PlatformStand = true
    Humanoid.AutoRotate = false

    local bv = Instance.new("BodyVelocity")
    bv.MaxForce = Vector3.new(4000, 4000, 4000)
    bv.Velocity = Vector3.zero
    bv.Parent = hrp
    ChillObjects.BodyVelocity = bv

    ChillActive = true
end

local function DisableChill()
    Humanoid.PlatformStand = false
    Humanoid.AutoRotate = true
    for _, obj in pairs(ChillObjects) do if obj then obj:Destroy() end end
    ChillActive = false
end

MainTab:AddToggle({
    Name = "ho√° chill guys",
    Default = false,
    Callback = function(state)
        if state then EnableChill() else DisableChill() end
    end
})
